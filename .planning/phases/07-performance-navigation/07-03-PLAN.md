```jsx
import { LineChart, Line, ResponsiveContainer } from 'recharts';

export default function Sparkline({ data, color = 'hsl(217, 91%, 60%)' }) {
  // data: [{ value: number }, ...]
  // If no data or single point, show flat line
  const chartData = (!data || data.length === 0)
    ? [{ value: 0 }, { value: 0 }]
    : data;

  return (
    <ResponsiveContainer width={48} height={24}>
      <LineChart data={chartData}>
        <Line
          type="monotone"
          dataKey="value"
          stroke={color}
          strokeWidth={1.5}
          dot={false}
          isAnimationActive={false}
        />
      </LineChart>
    </ResponsiveContainer>
  );
}
```

Key details:
- Fixed 48x24px via ResponsiveContainer (per UI-SPEC sparkline dimensions)
- No axes, no tooltip, no grid, no CartesianGrid
- `dot={false}` — no data point circles
- `isAnimationActive={false}` — instant render, no animation
- Stroke color = primary blue `hsl(217, 91%, 60%)` (CSS --primary value)
- Empty data gracefully handled with flat line

2. Add a `useMemo` after the `useEntriesData()` call to derive sparkline data per section. Use `entries` from context. Group entries by `section_name`, then for each section, derive weekly totals of `usage_amount` (per UI-SPEC — usage mode default). Normalize to `{ value }` shape:
   ```javascript
   const sparklineDataBySection = useMemo(() => {
     const bySectionName = {};
     for (const entry of entries) {
       if (!bySectionName[entry.section_name]) bySectionName[entry.section_name] = [];
       bySectionName[entry.section_name].push(entry);
     }
     const result = {};
     for (const [name, sectionEntries] of Object.entries(bySectionName)) {
       // Group by ISO week (Monday-based)
       const byWeek = {};
       for (const e of sectionEntries) {
         const d = new Date(e.date);
         const day = d.getDay() || 7;
         const monday = new Date(d);
         monday.setDate(d.getDate() - day + 1);
         const key = monday.toISOString().slice(0, 10);
         if (!byWeek[key]) byWeek[key] = 0;
         byWeek[key] += Number(e.usage_amount) || 0;
       }
       result[name] = Object.entries(byWeek)
         .sort(([a], [b]) => a.localeCompare(b))
         .slice(-7)  // last 7 weeks
         .map(([, val]) => ({ value: val }));
     }
     return result;
   }, [entries]);
   ```
