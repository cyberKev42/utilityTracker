```javascript
export function normalizeDecimal(value) {
  if (typeof value !== 'string') return String(value ?? '');
  return value.replace(',', '.');
}
```

5. Create `frontend/src/utils/sectionName.js`:
```javascript
export function getSectionDisplayName(section, t) {
  if (!section) return '';
  if (section.translation_key) {
    return t(`sectionNames.${section.translation_key}`, { defaultValue: section.name });
  }
  return section.name;
}
```

Note per D-08: When `translation_key` is NULL, always return `section.name` — do NOT call `t()` with a null key. The explicit `if (section.translation_key)` guard handles this (per Pitfall 3 in RESEARCH.md).
